	var faSubSp = faSpinner.init({
		'fontawesome_required' : true,
		'fontawesome_url' : 'https://ext.cdn-backslash.ch/lib/font-awesome/4.7/css/font-awesome.min.css',
		'btn_selector' : '[type=submit]',
		'timeout' : 100,
		'delay' : 400,
		'debugmode' : true,
		'ajax_listener' : true,
		'use_attribute_definition' : true
	});
	faSubSp.onload();
});
